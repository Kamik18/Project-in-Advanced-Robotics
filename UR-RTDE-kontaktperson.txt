import numpy as np

def fcn(R, P, tau, f):
    S = np.array([[0, -P[2], P[1]],
                  [P[2], 0, -P[0]],
                  [-P[1], P[0], 0]])

    Tau_ext = np.dot(R.T, tau)
    F_ext = -np.dot(R.T, np.dot(S, tau)) + np.dot(R.T, f)
    return F_ext, Tau_ext
